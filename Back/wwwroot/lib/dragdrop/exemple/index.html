Drag-drop jquery plugin

Simple

$('.block1')
  .dragdrop()
  .on('dragend', function (e, data) {
    $(this).css({left: data.position.x, top: data.position.y})
  });

Make clone

$('.block2')
  .dragdrop({
    makeClone: true
  })
  .on('dragend', function (e, data) {
    $(this).css({left: data.sourcePosition.x, top: data.sourcePosition.y})
  });

Wait start

$('.block2')
  .dragdrop({
    waitStart: 500,
    dragClass: 'drag'
  })
  .on('dragend', function (e, data) {
    $(this).css({left: data.sourcePosition.x, top: data.sourcePosition.y})
  });
